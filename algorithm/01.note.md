## [二分查找](./source_code/src/alg.binary_search.cc)

```c++
std::size_t binary_search(const int& key, const std::vector<int>& array) {
    std::size_t lo = 0;
    std::size_t hi = array.size() - 1;
    std::size_t mid = 0;
    while (lo <= hi) {
        mid = lo + (hi - lo) / 2;
        const int& mid_num = array[mid];
        if (key < mid_num) {
            hi = mid - 1;
        }
        else if (mid_num > key) {
            lo = mid + 1;
        }
        else {
            return mid;
        }
    }
    return -1;
}
```

## [Dijkstra 双栈算法](./source_code/src/alg.dijkstra_two_stack.cc)

```c++
double evaluate(const std::string& expression) {
    std::stack<std::string> ops;
    std::stack<double> vals;
    std::istringstream iss(expression);
    std::string s;
    while (iss >> s) {
        // 读取符号
        if (s == "(") {
            // 忽略左括号
            continue;
        }
        // 如果是运算符则压入栈
        if (s == "+") {
            ops.push(s);
        }
        else if (s == "-") {
            ops.push(s);
        }
        else if (s == "*") {
            ops.push(s);
        }
        else if (s == "/") {
            ops.push(s);
        }
        else if (s == "sqrt") {
            ops.push(s);
        }
        // 如果是右括号，取出运算符和操作数，计算结果，压入栈
        else if (s == ")") {
            std::string op = std::move(ops.top());
            ops.pop();
            double rv = vals.top();
            vals.pop();
            if (op == "+") {
                double lv = vals.top();
                vals.pop();
                rv += lv;
            }
            else if (op == "-") {
                double lv = vals.top();
                vals.pop();
                rv -= lv;
            }
            else if (op == "*") {
                double lv = vals.top();
                vals.pop();
                rv *= lv;
            }
            else if (op == "/") {
                double lv = vals.top();
                vals.pop();
                rv /= lv;
            }
            else if (op == "sqrt") {
                rv = std::sqrt(rv);
            }
            vals.push(rv);
        }
        // 不是符号，当作数字，转成 double ，压入栈
        else {
            vals.push(std::stod(s));
        }
    }
    // 最终运算结果
    return vals.top();
}
```
